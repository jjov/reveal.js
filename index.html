                        Map<String,Stack<MtnInfo>> connectionToPlansResponse = new HashMap<String,Stack<MtnInfo>> ()
                        for (def mtnInfo : mtnInfos ){
                            for ( def planInfo : mtnInfo.mtnPlanList.planInfo){
                                String deviceId = mtnInfo.eqptInfo.deviceInfo.deviceId? : mtnInfo.eqptInfo.deviceInfo.deviceSKU
                                String planId = planInfo.pricePlanId.split("~")[0]
                                String id = planId.concat(deviceId)
                                Stack<MtnInfo> mtns= connectionToPlansResponse.get(id) ?: new Stack<MtnInfo>()
                                mtns.push(mtnInfo)
                                connectionToPlansResponse.put(id,mtns)
                            }
                        }
                        return connectionToPlansResponse
                    }
